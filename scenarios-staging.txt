      <p><strong>Overview:</strong> Infinite scroll loads more data as the user approaches the bottom of the list, replacing traditional pagination. The core mechanism is detecting when a sentinel element (or the last item) enters the viewport using the <code>IntersectionObserver</code> API.</p>

      <p><strong>State structure:</strong></p>
      <ul>
        <li><code>items: T[]</code> — accumulated data</li>
        <li><code>page: number</code> — current page cursor</li>
        <li><code>hasMore: boolean</code> — whether the API has more data</li>
        <li><code>loading: boolean</code> — request in flight</li>
      </ul>

      <p><strong>Component hierarchy:</strong></p>
      <ul>
        <li><code>InfiniteList</code> — owns state, renders list + sentinel</li>
        <li><code>ListItem</code> — pure display component</li>
        <li><code>useInfiniteScroll</code> — custom hook encapsulating the observer logic</li>
      </ul>

      <p><strong>Key implementation:</strong></p>
<pre class="code-block">// useInfiniteScroll.ts
import { useEffect, useRef, useCallback } from 'react';

export function useInfiniteScroll(
  onLoadMore: () => void,
  hasMore: boolean,
  loading: boolean
) {
  const sentinelRef = useRef&lt;HTMLDivElement | null&gt;(null);

  const handleObserver = useCallback(
    (entries: IntersectionObserverEntry[]) => {
      const [entry] = entries;
      if (entry.isIntersecting && hasMore && !loading) {
        onLoadMore();
      }
    },
    [onLoadMore, hasMore, loading]
  );

  useEffect(() => {
    const el = sentinelRef.current;
    if (!el) return;
    const observer = new IntersectionObserver(handleObserver, {
      root: null,
      rootMargin: '200px', // trigger 200px before bottom
      threshold: 0,
    });
    observer.observe(el);
    return () => observer.disconnect();
  }, [handleObserver]);

  return sentinelRef;
}

// InfiniteList.tsx
export function InfiniteList() {
  const [items, setItems] = useState&lt;Item[]&gt;([]);
  const [page, setPage] = useState(1);
  const [hasMore, setHasMore] = useState(true);
  const [loading, setLoading] = useState(false);

  const loadMore = useCallback(async () => {
    setLoading(true);
    const data = await fetchItems(page);
    setItems(prev => [...prev, ...data.items]);
    setHasMore(data.hasMore);
    setPage(p => p + 1);
    setLoading(false);
  }, [page]);

  const sentinelRef = useInfiniteScroll(loadMore, hasMore, loading);

  useEffect(() => { loadMore(); }, []); // initial load

  return (
    &lt;div&gt;
      {items.map(item =&gt; &lt;ListItem key={item.id} item={item} /&gt;)}
      {loading &amp;&amp; &lt;Spinner /&gt;}
      &lt;div ref={sentinelRef} style={{ height: 1 }} /&gt;
    &lt;/div&gt;
  );
}</pre>

      <p><strong>Performance considerations:</strong></p>
      <ul>
        <li>Virtualise the list with <code>react-window</code> or <code>react-virtual</code> when item count grows large to avoid DOM bloat.</li>
        <li>Debounce or gate <code>loadMore</code> with the <code>loading</code> flag to prevent double-fetches.</li>
        <li>Use <code>useCallback</code> on <code>loadMore</code> so the observer is not recreated on every render.</li>
        <li>For server state, prefer <strong>TanStack Query</strong> (<code>useInfiniteQuery</code>) which handles caching, deduplication, and background refetch automatically.</li>
      </ul>

      <p><strong>Libraries to consider:</strong> <code>react-query</code> / TanStack Query (useInfiniteQuery), <code>react-window</code>, <code>react-intersection-observer</code> (wrapper around IntersectionObserver).</p>

  SortableContext, verticalListSortingStrategy,
  useSortable, arrayMove
} from '@dnd-kit/sortable';
import { CSS } from '@dnd-kit/utilities';

function SortableItem({ id, label }: { id: string; label: string }) {
  const { attributes, listeners, setNodeRef, transform, transition, isDragging } =
    useSortable({ id });

  const style = {
    transform: CSS.Transform.toString(transform),
    transition,
    opacity: isDragging ? 0.5 : 1,
  };

  return (
    &lt;div ref={setNodeRef} style={style} {...attributes} {...listeners}
      className="sortable-item"&gt;
      {label}
    &lt;/div&gt;
  );
}

export function SortableList() {
  const [items, setItems] = useState([
    { id: '1', label: 'Item A' },
    { id: '2', label: 'Item B' },
    { id: '3', label: 'Item C' },
  ]);

  const sensors = useSensors(useSensor(PointerSensor));

  function handleDragEnd(event: DragEndEvent) {
    const { active, over } = event;
    if (over &amp;&amp; active.id !== over.id) {
      setItems(prev =&gt; {
        const oldIndex = prev.findIndex(i =&gt; i.id === active.id);
        const newIndex = prev.findIndex(i =&gt; i.id === over.id);
        return arrayMove(prev, oldIndex, newIndex);
      });
    }
  }

  return (
    &lt;DndContext sensors={sensors} collisionDetection={closestCenter}
      onDragEnd={handleDragEnd}&gt;
      &lt;SortableContext items={items.map(i =&gt; i.id)}
        strategy={verticalListSortingStrategy}&gt;
        {items.map(item =&gt; (
          &lt;SortableItem key={item.id} id={item.id} label={item.label} /&gt;
        ))}
      &lt;/SortableContext&gt;
    &lt;/DndContext&gt;
  );
}</pre>

<pre class="code-block">// useWebSocket.ts
import { useEffect, useRef, useState, useCallback } from 'react';

export function useWebSocket&lt;T&gt;(url: string) {
  const ws = useRef&lt;WebSocket | null&gt;(null);
  const [data, setData] = useState&lt;T | null&gt;(null);
  const [status, setStatus] = useState&lt;'connecting'|'open'|'closed'&gt;('connecting');

  const connect = useCallback(() => {
    ws.current = new WebSocket(url);
    ws.current.onopen = () => setStatus('open');
    ws.current.onmessage = (e) => setData(JSON.parse(e.data) as T);
    ws.current.onclose = () => {
      setStatus('closed');
      setTimeout(connect, 3000); // reconnect after 3s
    };
  }, [url]);

  useEffect(() => {
    connect();
    return () => {
      ws.current?.close();
    };
  }, [connect]);

  const send = useCallback((msg: unknown) => {
    ws.current?.send(JSON.stringify(msg));
  }, []);

  return { data, status, send };
}

// Dashboard.tsx
function LiveDashboard() {
  const { data: metrics } = useWebSocket&lt;DashboardMetrics&gt;(
    'wss://api.example.com/dashboard'
  );

  return (
    &lt;div&gt;
      &lt;MetricCard label="Active Users" value={metrics?.activeUsers} /&gt;
      &lt;MetricCard label="Orders/min" value={metrics?.ordersPerMin} /&gt;
    &lt;/div&gt;
  );
}</pre>

      <p><strong>State management at scale:</strong> For multiple subscribers to the same WS stream, lift the socket to a Context provider so the connection is shared. Dispatch incoming messages to a Redux slice or Zustand store so any component can subscribe.</p>

      <p><strong>Performance considerations:</strong></p>
      <ul>
        <li>Throttle or debounce high-frequency updates (e.g., max 60fps) using <code>requestAnimationFrame</code> or a time-based gate.</li>
        <li>Use <code>React.memo</code> and selector granularity to avoid re-rendering the entire dashboard on every message.</li>
        <li>For charts, update data via refs and call the chart library's own update method rather than triggering a React re-render.</li>
      </ul>

  useEffect(() =&gt; {
    const id = setTimeout(() =&gt; setDebounced(value), delay);
    return () =&gt; clearTimeout(id);
  }, [value, delay]);

  past: T[];
  present: T;
  future: T[];

function useHistory&lt;T&gt;(initial: T) {
  const [history, setHistory] = useState&lt;HistoryState&lt;T&gt;&gt;({
    past: [], present: initial, future: []
  });

  const set = useCallback((newPresent: T) =&gt; {
    setHistory(h =&gt; ({
      past: [...h.past, h.present],
      present: newPresent,
      future: [], // clear redo stack on new action
    }));
  }, []);

  const undo = useCallback(() =&gt; {
    setHistory(h =&gt; {
      if (h.past.length === 0) return h;
      const previous = h.past[h.past.length - 1];
      return {
        past: h.past.slice(0, -1),
        present: previous,
        future: [h.present, ...h.future],
      };
    });
  }, []);

  const redo = useCallback(() =&gt; {
    setHistory(h =&gt; {
      if (h.future.length === 0) return h;
      const next = h.future[0];
      return {
        past: [...h.past, h.present],
        present: next,
        future: h.future.slice(1),
      };
    });
  }, []);

  return { state: history.present, set, undo, redo,
           canUndo: history.past.length &gt; 0,
           canRedo: history.future.length &gt; 0 };
}

    useHistory&lt;Stroke[]&gt;([]);

      if ((e.ctrlKey || e.metaKey) &amp;&amp; e.key === 'z') {
        e.shiftKey ? redo() : undo();
      }

      <p><strong>State sharing strategies:</strong></p>
      <ul>
        <li><strong>Shared Redux store via Module Federation</strong> — strong coupling, but real-time sync.</li>
        <li><strong>Custom Events / postMessage</strong> — loose coupling; MFEs broadcast events (<code>window.dispatchEvent(new CustomEvent('cart:updated', { detail }))</code>) and others listen. No shared library needed.</li>
        <li><strong>URL as shared state</strong> — route params visible to all MFEs loaded on the same page.</li>
        <li><strong>Shared BFF (Backend for Frontend)</strong> — MFEs independently fetch from the same server, relying on server state as the source of truth. No client-side sharing needed.</li>
      </ul>

      <p><strong>Framework-agnostic communication (recommended for heterogeneous MFEs):</strong> Use a tiny publish/subscribe event bus exposed on a shared global object (<code>window.__eventBus</code>) or injected via Module Federation. Each MFE subscribes to topics it cares about without knowing about others.</p>

      <p><strong>Routing:</strong> The Shell owns the router. Each MFE receives its base path as a prop and renders its own sub-routes using React Router's <code>basename</code> option.</p>

      <p><strong>Deployment:</strong> Each MFE is deployed independently to its own CDN path. The shell fetches the remote entry URLs from a manifest service so new MFE versions are picked up without redeploying the shell.</p>
