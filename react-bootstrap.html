Three setup options, real-world component patterns, and a complete Bootstrap 5 utility class cheat sheet.

Setup — 3 Options

# Option 1 — Bootstrap CSS + JS (simplest, use Bootstrap classes directly)
npm install bootstrap bootstrap-icons
# In your main entry file (main.jsx or index.js):
import 'bootstrap/dist/css/bootstrap.min.css';
import 'bootstrap/dist/js/bootstrap.bundle.min.js';
import 'bootstrap-icons/font/bootstrap-icons.css';

// Then use Bootstrap classes directly in JSX:
function MyButton() {
  return <button className="btn btn-primary">Click me</button>;
}
Best for: Most React projects. You get the full Bootstrap utility system. JavaScript components (modals, dropdowns) work via data attributes or Bootstrap's JS API.
# Option 2 — React Bootstrap (proper React component library)
npm install react-bootstrap bootstrap
# Import CSS in main.jsx, then use components:
import { Button, Modal, Form, Table, Card, Nav,
         Navbar, Badge, Alert, Spinner } from 'react-bootstrap';

function UserForm({ show, onHide, onSubmit }) {
  const [name, setName] = useState('');
  const [email, setEmail] = useState('');

  return (
    <Modal show={show} onHide={onHide} centered>
      <Modal.Header closeButton>
        <Modal.Title>Add User</Modal.Title>
      </Modal.Header>
      <Modal.Body>
        <Form>
          <Form.Group className="mb-3">
            <Form.Label>Name</Form.Label>
            <Form.Control
              value={name}
              onChange={e => setName(e.target.value)}
              placeholder="Enter name"
            />
          </Form.Group>
          <Form.Group className="mb-3">
            <Form.Label>Email</Form.Label>
            <Form.Control
              type="email"
              value={email}
              onChange={e => setEmail(e.target.value)}
            />
          </Form.Group>
        </Form>
      </Modal.Body>
      <Modal.Footer>
        <Button variant="secondary" onClick={onHide}>Cancel</Button>
        <Button variant="primary" onClick={() => onSubmit({ name, email })}>
          Save
        </Button>
      </Modal.Footer>
    </Modal>
  );
}
Best for: Projects where you want Bootstrap components (Modal, Dropdown, Tooltip) to work as proper React components with props — no jQuery or Bootstrap JS manipulation needed.

Dynamic Sortable Table

A reusable table component with column-based sorting, sort direction indicator, and Bootstrap table styles.

function DataTable({ columns, data, onSort }) {
  const [sortField, setSortField] = useState(null);
  const [sortDir, setSortDir] = useState('asc');

  const handleSort = (field) => {
    const dir = sortField === field && sortDir === 'asc' ? 'desc' : 'asc';
    setSortField(field);
    setSortDir(dir);
    onSort?.(field, dir);
  };

  return (
    <div className="table-responsive">
      <table className="table table-hover table-bordered align-middle">
        <thead className="table-dark">
          <tr>
            {columns.map(col => (
              <th
                key={col.field}
                onClick={() => handleSort(col.field)}
                style={{ cursor: 'pointer', userSelect: 'none' }}
              >
                {col.label}
                {sortField === col.field && (
                  <i className={`bi bi-sort-${sortDir === 'asc' ? 'down' : 'up'} ms-1`} />
                )}
              </th>
            ))}
          </tr>
        </thead>
        <tbody>
          {data.map((row, i) => (
            <tr key={row.id || i}>
              {columns.map(col => (
                <td key={col.field}>
                  {col.render ? col.render(row[col.field], row) : row[col.field]}
                </td>
              ))}
            </tr>
          ))}
          {data.length === 0 && (
            <tr>
              <td colSpan={columns.length} className="text-center text-muted py-4">
                No data found
              </td>
            </tr>
          )}
        </tbody>
      </table>
    </div>
  );
}

// Usage example
const columns = [
  { field: 'name', label: 'Name' },
  { field: 'email', label: 'Email' },
  {
    field: 'status',
    label: 'Status',
    render: (val) => (
      <span className={`badge bg-${val === 'active' ? 'success' : 'secondary'}`}>
        {val}
      </span>
    )
  },
];

Toast Notification System

A custom hook + container for Bootstrap-styled toast notifications that auto-dismiss after 3 seconds.

// Custom hook — returns the toasts array and a show() function
function useToast() {
  const [toasts, setToasts] = useState([]);

  const show = useCallback((message, type = 'success') => {
    const id = Date.now();
    setToasts(prev => [...prev, { id, message, type }]);
    setTimeout(() => {
      setToasts(prev => prev.filter(t => t.id !== id));
    }, 3000);
  }, []);

  const dismiss = useCallback((id) => {
    setToasts(prev => prev.filter(t => t.id !== id));
  }, []);

  return { toasts, show, dismiss };
}

// Toast container — renders in top-right corner
function ToastContainer({ toasts, onDismiss }) {
  return (
    <div
      className="position-fixed top-0 end-0 p-3"
      style={{ zIndex: 9999 }}
    >
      {toasts.map(toast => (
        <div
          key={toast.id}
          className={`toast show align-items-center text-white bg-${toast.type} border-0 mb-2`}
          role="alert"
        >
          <div className="d-flex">
            <div className="toast-body">{toast.message}</div>
            <button
              type="button"
              className="btn-close btn-close-white me-2 m-auto"
              onClick={() => onDismiss(toast.id)}
            />
          </div>
        </div>
      ))}
    </div>
  );
}

// Usage in your app
function App() {
  const { toasts, show, dismiss } = useToast();

  return (
    <>
      <ToastContainer toasts={toasts} onDismiss={dismiss} />
      <button onClick={() => show('Saved successfully!', 'success')}>Save</button>
      <button onClick={() => show('Failed to load data', 'danger')}>Fail</button>
      <button onClick={() => show('Processing...', 'warning')}>Warn</button>
    </>
  );
}
Production tip: For larger apps, extract the toast state into a Context + Provider so any component can call show() without prop drilling.

Responsive Card Grid

A fully responsive grid using Bootstrap's column system — 1 column on mobile, 2 on tablet, 3 on desktop, 4 on xl.

function CardGrid({ items, onSelect }) {
  return (
    <div className="row g-4">  {/* g-4 = gap between all cards */}
      {items.map(item => (
        <div key={item.id} className="col-sm-6 col-lg-4 col-xl-3">
          {/* col-sm-6 = 2 cols on sm+, col-lg-4 = 3 cols on lg+, col-xl-3 = 4 cols on xl+ */}
          <div
            className="card h-100 shadow-sm border-0"
            style={{ cursor: 'pointer', transition: 'transform .15s' }}
            onClick={() => onSelect?.(item)}
            onMouseEnter={e => e.currentTarget.style.transform = 'translateY(-4px)'}
            onMouseLeave={e => e.currentTarget.style.transform = 'none'}
          >
            {item.image && (
              <img
                src={item.image}
                className="card-img-top"
                alt={item.title}
                style={{ height: '200px', objectFit: 'cover' }}
              />
            )}
            <div className="card-body d-flex flex-column">
              <div className="d-flex justify-content-between align-items-start mb-2">
                <h5 className="card-title mb-0">{item.title}</h5>
                {item.badge && (
                  <span className="badge bg-primary ms-2">{item.badge}</span>
                )}
              </div>
              <p className="card-text text-muted flex-grow-1 small">{item.description}</p>
              <a href={item.link} className="btn btn-primary mt-auto">
                Learn More <i className="bi bi-arrow-right ms-1" />
              </a>
            </div>
          </div>
        </div>
      ))}
    </div>
  );
}

Dark Mode Toggle with Bootstrap 5.3

Bootstrap 5.3 introduced native dark mode via data-bs-theme. This component syncs React state, the DOM attribute, and localStorage.

// Custom hook — handles all the logic
function useDarkMode() {
  const [dark, setDark] = useState(() => {
    const stored = localStorage.getItem('theme');
    if (stored) return stored === 'dark';
    // Default to user's OS preference
    return window.matchMedia('(prefers-color-scheme: dark)').matches;
  });

  useEffect(() => {
    document.documentElement.setAttribute('data-bs-theme', dark ? 'dark' : 'light');
    localStorage.setItem('theme', dark ? 'dark' : 'light');
  }, [dark]);

  const toggle = useCallback(() => setDark(d => !d), []);

  return { dark, toggle };
}

// Button component
function DarkModeToggle() {
  const { dark, toggle } = useDarkMode();

  return (
    <button
      className="btn btn-outline-secondary"
      onClick={toggle}
      title={dark ? 'Switch to light mode' : 'Switch to dark mode'}
      aria-label="Toggle dark mode"
    >
      <i className={`bi bi-${dark ? 'sun' : 'moon-stars'}-fill`} />
      <span className="ms-2 d-none d-sm-inline">{dark ? 'Light' : 'Dark'}</span>
    </button>
  );
}

// Add to your Navbar:
// <DarkModeToggle />
Bootstrap 5.3 Dark Mode: When data-bs-theme="dark" is set on <html>, Bootstrap automatically adjusts all component colors — cards, tables, navbars, inputs. No custom CSS needed for standard components.
